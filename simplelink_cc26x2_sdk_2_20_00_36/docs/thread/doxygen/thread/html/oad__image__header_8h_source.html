127 /*
128  * The complete image header is a combination of:
129  * First: struct img_common_header
130  * Followed by 1 or more, in any order:
131  * struct img_seg_payload
132  * struct img_seg_boundary
133  * struct img_seg_signature
134  * The last item is always:
135  * struct img_seg_last
136  *
137  * A typical true header might be as follows
138  * struct my_header {
139  * struct img_common_header header;
140  * struct img_seg_payload payload;
141  * struct img_seg_last last;
142  * };
143  *
144  * Or
145  * struct my_header {
146  * struct img_common_header header;
147  * struct img_seg_boundary boundary;
148  * struct img_seg_payload payload;
149  * struct img_seg_security security;
150  * struct img_seg_last last;
151  * };
152  */
