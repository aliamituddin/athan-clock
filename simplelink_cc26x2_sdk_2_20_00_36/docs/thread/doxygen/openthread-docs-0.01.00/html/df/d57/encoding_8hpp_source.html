68 inline uint32_t Reverse32(uint32_t v)
69 {
70  v = ((v & 0x55555555) << 1) | ((v & 0xaaaaaaaa) >> 1);
71  v = ((v & 0x33333333) << 2) | ((v & 0xcccccccc) >> 2);
72  v = ((v & 0x0f0f0f0f) << 4) | ((v & 0xf0f0f0f0) >> 4);
73  v = ((v & 0x00ff00ff) << 8) | ((v & 0xff00ff00) >> 8);
74  v = ((v & 0x0000ffff) << 16) | ((v & 0xffff0000) >> 16);
75 
76  return v;
77 }
