191 private:
192  /*
193  * The RssAverager uses an adaptive exponentially weighted filter to maintain the average. It keeps track of
194  * current average and the number of added RSS values (up to a 8).
195  *
196  * For the first 8 added RSS values, the average is the arithmetic mean of the added values (e.g., if 5 values are
197  * added, the average is sum of the 5 added RSS values divided by 5. After the 8th RSS value, a weighted filter is
198  * used with coefficients (1/8, 7/8), i.e., newAverage = 1/8 * newRss + 7/8 * oldAverage.
199  *
200  * To add to accuracy of the averaging process, the RSS values and the maintained average are multiplied by a
201  * precision factor of -8.
202  *
203  */
204 
205  enum
206  {
207  kPrecisionBitShift = 3, // Precision multiple for RSS average (1 << PrecisionBitShift).
208  kPrecision = (1 << kPrecisionBitShift),
209  kPrecisionBitMask = (kPrecision - 1),
210 
211  kCoeffBitShift = 3, // Coefficient used for exponentially weighted filter (1 << kCoeffBitShift).
212  };
213 
214  // Member variables fit into two bytes.
215 
216  uint16_t mAverage : 11; // The raw average signal strength value (stored as RSS times precision multiple).
217  uint16_t mCount : 5; // Number of RSS values added to averager so far (limited to 2^kCoeffBitShift-1).
218 };
