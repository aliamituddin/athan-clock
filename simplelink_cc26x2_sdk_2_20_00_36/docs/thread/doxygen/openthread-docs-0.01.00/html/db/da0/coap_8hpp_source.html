68  kAckRandomFactorNumerator = OPENTHREAD_CONFIG_COAP_ACK_RANDOM_FACTOR_NUMERATOR,
69  kAckRandomFactorDenominator = OPENTHREAD_CONFIG_COAP_ACK_RANDOM_FACTOR_DENOMINATOR,
70  kMaxRetransmit = OPENTHREAD_CONFIG_COAP_MAX_RETRANSMIT,
71  kNStart = 1,
72  kDefaultLeisure = 5,
73  kProbingRate = 1,
74 
75  // Note that 2 << (kMaxRetransmit - 1) is equal to kMaxRetransmit power of 2
76  kMaxTransmitSpan =
77  kAckTimeout * ((2 << (kMaxRetransmit - 1)) - 1) * kAckRandomFactorNumerator / kAckRandomFactorDenominator,
78  kMaxTransmitWait =
79  kAckTimeout * ((2 << kMaxRetransmit) - 1) * kAckRandomFactorNumerator / kAckRandomFactorDenominator,
80  kMaxLatency = 100,
81  kProcessingDelay = kAckTimeout,
82  kMaxRtt = 2 * kMaxLatency + kProcessingDelay,
83  kExchangeLifetime = kMaxTransmitSpan + 2 * (kMaxLatency) + kProcessingDelay,
84  kNonLifetime = kMaxTransmitSpan + kMaxLatency
85 };
